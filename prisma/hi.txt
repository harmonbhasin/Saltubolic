generator client {
  provider = "prisma-client-js"
}

datasource db {
  provider     = "mysql"
  url          = env("DATABASE_URL")
  relationMode = "prisma"
}

model Client {
  id         String   @id @default(uuid())
  createdAt  DateTime @default(now())
  updatedAt  DateTime @updatedAt
  profileId  String   @unique
  personalId String   @unique
  notes      Note[]
  profile    Profile  @relation(fields: [profileId], references: [id])
  personal   Personal @relation(fields: [personalId], references: [id])

  @@index([profileId])
  @@index([personalId])
}

model Profile {
  id           String   @id @default(uuid())
  createdAt    DateTime @default(now())
  updatedAt    DateTime @updatedAt
  programName  String
  workout      String
  dailyMessage Boolean

  goal         goalType @default(MAINTAIN)
  intensity    intensityType @default(MODERATE)
  snapshotFrequency snapshotFrequencyType @default(WEEKLY)

  weightUsage Int      @default(0)
  scheduleUsage Int    @default(0)


  client       Client?
}

model Personal {
  id        String    @id @default(uuid())
  createdAt DateTime  @default(now())
  updatedAt DateTime  @updatedAt
  fname     String
  lname     String
  email     String    @unique
  phone     String    @unique
  address   Address[]
  client    Client?
}

model Address {
  id         String      @id @default(uuid())
  createdAt  DateTime    @default(now())
  updatedAt  DateTime    @updatedAt
  addressOne String
  addressTwo String
  city       String
  state      String
  zipcode    String
  type       addressType @default(HOME)
  personalId String?
  personal   Personal?   @relation(fields: [personalId], references: [id])

  @@index([personalId])
}

model Note {
  id        String   @id @default(uuid())
  createdAt DateTime @default(now())
  frequency noteType @default(WEEKLY)
  note      String
  sentiment String?
  clientId  String
  author    Client   @relation(fields: [clientId], references: [id])

  @@index([clientId])
}

enum addressType {
  HOME
  WORK
}

enum noteType {
  WEEKLY
  DAILY
  MONTHLY
}

enum goalType {
  BULK
  CUT
  MAINTAIN
}

enum intensityType {
  KIND
  MODERATE
  HARSH
}

enum snapshotFrequencyType {
  WEEKLY
  MONTHLY
}
